$constcoder = @{
    name = 'Harsha Rajoor'
    skills = @('Python', 'PowerShell', 'Bash', 'Assembly language', 'C', 'HTML', 'AWS')
    hardWorker = $true
    quickLearner = $true
    problemSolver = $true
}

function Hireable {
    param (
        [hashtable]$coder
    )

    return (
        $coder.hardWorker -and
        $coder.problemSolver -and
        $coder.skills.Count -ge 5
    )
}

$hireableStatus = Hireable -coder $constcoder
Write-Output $hireableStatus
True
